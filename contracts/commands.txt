// Deploying to Ethereum Sepolia

forge script script/Deploy.s.sol:DeployDecentralizedIssueTracker \
  --rpc-url sepolia \
  --broadcast \
  --verify \
  --etherscan-api-key $ETHERSCAN_API_KEY \
  -vvvv

// Alternative deployment without verification (faster)
forge script script/Deploy.s.sol:DeployDecentralizedIssueTracker \
  --rpc-url sepolia \
  --broadcast \
  -vvvv

// Manual verification after deployment
forge verify-contract \
  <CONTRACT_ADDRESS> \
  src/DecentralizedIssueTracker.sol:DecentralizedIssueTracker \
  --rpc-url sepolia \
  --etherscan-api-key $ETHERSCAN_API_KEY \
  --constructor-args "0x000000000000000000000000000000000000dead" \
  -vvvv

// Using public RPC (no API key needed)
forge script script/Deploy.s.sol:DeployDecentralizedIssueTracker \
  --rpc-url https://rpc.sepolia.org \
  --broadcast \
  --verify \
  --etherscan-api-key $ETHERSCAN_API_KEY \
  -vvvv
